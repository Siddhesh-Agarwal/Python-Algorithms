Sorting Techniques
  1. Bubble Sort
    2. def bubble_sort(arr):
    for iter_num in range(len(arr) - 1, 0, -1):
        for i in range(iter_num):
            if arr[i] > arr[i + 1]:
                arr[i], arr[i + 1] = arr[i + 1], arr[i]
  2. Insertion Sort
    3. def insertion_sort(arr):
    for i in range(1, len(arr)):                  
        key = arr[i]
        j = i - 1
        while j >= 0 and key < arr[j] :
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
  3. Shell Sort
    4. def shell_sort(arr):
    gap = len(arr) // 2
    while gap > 0:
        for i in range(gap, len(arr)):
            temp = arr[i]
            j = i
    while j >= gap and arr[j - gap] > temp:
        arr[j] = arr[j - gap]
        j = j-gap
        arr[j] = temp
    gap = gap//2
  4. Counting Sort
    5. def counting_sort(arr):
    maxEl = max(arr)
    countArrayLength = maxEl+1
    countArray = [0] * countArrayLength

    for el in arr: 
        countArray[el] += 1

    for i in range(1, countArrayLength):
        countArray[i] += countArray[i-1]

    outputArray = [0] * len(arr)
    i = len(arr) - 1

    while i >= 0:
        currentEl = arr[i]
        countArray[currentEl] -= 1
        newPosition = countArray[currentEl]
        outputArray[newPosition] = currentEl
        i -= 1

    return outputArray
Searching Techniques
  1. Linear search
    2. # Linear Search - time complexity: O(n)
def LinearSearch(array, element):
    for i in range(len(array)):
        if arr[i] == element:
            return i
    return -1
  2. Binary search
    3. # Binary Search - time complexity: O(log(n) base 2)
def BinarySearch(array, element):
    LOW = 0
    TOP = len(array)-1
    while LOW <= TOP:
        MID = (TOP + LOW)//2
        if element > array[MID]:
            LOW += 1
        elif element < array[MID]:
            TOP -= 1
        else:
            return MID
    return -1
